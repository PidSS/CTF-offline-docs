    for x, cnt in r:
        for _ in range(cnt):
            lis.append(x)
    
    print(f'factor: {lis}')
    
    ans = []
    
    for num in range(len(lis)):
        for f in itertools.combinations(lis, num):

                maybe_s = functools.reduce(lambda x,y: x*y, f)
                
                if 0xe000000000000003 <= maybe_s <= 2**64:
                    ans.append(maybe_s)
    
    return ans
