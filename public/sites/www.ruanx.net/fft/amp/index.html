import numpy as np

PI = 3.1415926535

def fft(A):
    n = len(A)
    
    if n==1:
        return [A[0]]
    
    A0 = fft(A[0::2])
    A1 = fft(A[1::2])
    
    res = np.zeros(n, dtype = np.complex)
    
    omega = np.cos(2.0*PI/n) + np.sin(2.0*PI/n)*1.0j
    x = 1
    
    for k in range(n//2):        
        res[k] = A0[k] + x * A1[k]
        res[k+n//2] = A0[k] - x * A1[k]
    
        x *= omega
    
    return res

def ifft(A):
    n = len(A)
    
    if n==1:
        return [A[0]]
    
    A0 = ifft(A[0::2])
    A1 = ifft(A[1::2])
    
    res = np.zeros(n, dtype = np.complex)
    
    omega = np.cos(2.0*PI/n) - np.sin(2.0*PI/n)*1.0j
    x = 1
    
    for k in range(n//2):
        res[k] = A0[k] + x * A1[k]
        res[k+n//2] = A0[k] - x * A1[k]
        
        x *= omega
    
    return res

def times(A, B):
    lenC = len(A) + len(B) - 1
    pts = 1
    
    while pts < lenC:
        pts *= 2
    
    X = np.zeros(pts, dtype=float)
    X[0:len(A)] = A
    Y = np.zeros(pts, dtype=float)
    Y[0:len(B)] = B
    
    Z = ifft(fft(X) * fft(Y))[:lenC] / pts
    
    print(' '.join([str(int(x)) for x in Z.real.round()]))
